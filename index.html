前端学习博客 arguments 不能人为创建，只能在函数创建时候生成 arguments对象包含了如下属性： callee —— 对当前函数的引用 length —— 实参的个数 arguments中的参数和函数的实际传递的参数是共享的 关于arguments的代码测试如下： function testAru(a,b,c){

//函数定义的参数个数
alert(testAru.length);

//实际传递的参数个数
alert(arguments.length);

//引用函数自身 可以用于实现匿名递归函数
alert(arguments.callee === testAru);

//参数相互共享
alert(a === arguments[0]);
alert(a);

a = 30;
alert(arguments[0]);

//然而对于没有传递的参数c,参数不不共享的
c = 40;
alert(arguments[2]);

arguments[2] = 50;
alert(c);

//依次访问函数参数
for(var i=0;i<arguments.length;i++){
    alert(arguments[i]);//a = 30 b =20
}

}

testAru(10,20);
